-- ANTI-ANTICHEAT TELEPORT SYSTEM
local function SafeTeleport(targetCFrame)
    local char = LocalPlayer.Character
    if not char then return end
    
    local root = char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("Torso")
    if not root then return end
    
    ShowNotification("âš¡ Teleporting...", 2)
    
    if AntiCheatBypass then
        -- METHOD 1: Gradual teleport (undetectable)
        if GhostMode then
            local startPos = root.Position
            local endPos = targetCFrame.Position
            local distance = (endPos - startPos).Magnitude
            
            -- Create fake parts at old position
            local fakePart = Instance.new("Part")
            fakePart.Size = Vector3.new(4, 6, 2)
            fakePart.CFrame = root.CFrame
            fakePart.Transparency = 1
            fakePart.CanCollide = false
            fakePart.Anchored = true
            fakePart.Parent = workspace
            
            -- Gradual teleport in small steps
            local steps = math.ceil(distance / 10)
            for i = 1, steps do
                local progress = i / steps
                local newPos = startPos:Lerp(endPos, progress)
                root.CFrame = CFrame.new(newPos) * CFrame.Angles(0, root.CFrame.LookVector.Y, 0)
                task.wait(0.01) -- Small delay to avoid detection
            end
            
            fakePart:Destroy()
            
        -- METHOD 2: Below-ground teleport
        elseif targetCFrame.Position.Y > -100 then
            -- Teleport below ground first
            local belowGround = CFrame.new(targetCFrame.Position - Vector3.new(0, 50, 0))
            root.CFrame = belowGround
            
            task.wait(0.1)
            
            -- Slowly move up to target
            for i = 1, 10 do
                local newY = belowGround.Position.Y + (i * 5)
                root.CFrame = CFrame.new(targetCFrame.Position.X, newY, targetCFrame.Position.Z)
                task.wait(0.05)
            end
            
            -- Final position
            root.CFrame = targetCFrame
            
        -- METHOD 3: Network ownership + teleport
        else
            -- Take network ownership first
            pcall(function()
                root:SetNetworkOwner(LocalPlayer)
                for _, part in ipairs(char:GetDescendants()) do
                    if part:IsA("BasePart") then
                        pcall(function() part:SetNetworkOwner(LocalPlayer) end)
                    end
                end
            end)
            
            task.wait(0.05)
            root.CFrame = targetCFrame
        end
        
    else
        -- Direct teleport (might get detected)
        root.CFrame = targetCFrame
    end
    
    -- Reset velocity to avoid rubberbanding
    root.Velocity = Vector3.new(0, 0, 0)
    root.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
    root.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
    
    ShowNotification("âœ… Teleported!", 1)
    LastSafePosition = targetCFrame
end

-- HOTKEYS FOR PC (optional)
local function SetupHotkeys()
    UserInputService.InputBegan:Connect(function(input, gameProcessed)
        if gameProcessed then return end
        
        -- F1: Save Point A
        if input.KeyCode == Enum.KeyCode.F1 then
            QuickSavePosition("PointA")
        end
        
        -- F2: Teleport to Point A
        if input.KeyCode == Enum.KeyCode.F2 then
            if SavedPositions.PointA then
                SafeTeleport(SavedPositions.PointA)
            end
        end
        
        -- F3: Toggle GUI
        if input.KeyCode == Enum.KeyCode.F3 then
            local gui = LocalPlayer.PlayerGui:FindFirstChild("MobileTeleportGUI")
            if gui then
                gui.Enabled = not gui.Enabled
                ShowNotification("GUI: " .. (gui.Enabled and "Shown" or "Hidden"), 2)
            end
        end
    end)
end
